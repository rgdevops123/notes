NGINX:  HTTP Load Balancer


$ sudo yum -y install nginx

$ sudo vim /etc/nginx/nginx/nginx.conf
+++
http {
    upstream app {
        server 10.0.0.96:5000;
        server 10.0.0.97:5000;
        server 10.0.0.98:5000;
    }

    server {
        listen 80;

        location / {
            proxy_pass http://app;
        }
    }
}
+++


$ sudo systemctl restart nginx.service
